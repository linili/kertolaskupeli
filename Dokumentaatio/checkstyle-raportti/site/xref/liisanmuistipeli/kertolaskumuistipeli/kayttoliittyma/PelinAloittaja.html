6   package liisanmuistipeli.kertolaskumuistipeli.kayttoliittyma;
7   
8   import liisanmuistipeli.kertolaskumuistipeli.logiikka.Logiikka;
9   import liisanmuistipeli.kertolaskumuistipeli.logiikka.KorttienArpoja;
10  import liisanmuistipeli.kertolaskumuistipeli.logiikka.Pelitilanne;
11  
12  /**
13   *
14   * @author liisapauliina
15   */
16  public class PelinAloittaja {
17  
18      private Logiikka logiikka;
19      private KorttienArpoja arpoja;
20      private GraafinenKayttoliittyma kali;
21  
22      public PelinAloittaja() {
23      }
24  
25      public int kysyKertoja() {
26          int kertoja = VaihtoehtoinenAloitusvalikko.kysyKertoja("Syötä kertotaulu (1-9) ");
27          while (kertoja < 1 || kertoja > 9) {
28              kertoja = VaihtoehtoinenAloitusvalikko.kysyKertoja("Syötä kertotaulu väliltä 1-9 ");
29          }
30          return kertoja;
31      }
32  
33      public void teeUusiPeli(int kertoja) {
34          this.arpoja = new KorttienArpoja();
35          this.logiikka = new Logiikka(this.arpoja.luoSatunnainenPelitilanne(kertoja));
36          this.kali = new GraafinenKayttoliittyma(this.logiikka);
37          this.kali.run();
38      }
39  
40      public Logiikka getLogiikka() {
41          return this.logiikka;
42      }
43  }
