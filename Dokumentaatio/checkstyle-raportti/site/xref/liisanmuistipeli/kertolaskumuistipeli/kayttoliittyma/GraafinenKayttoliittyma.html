24  /**
25   * Luokka toimii graafisena käyttöliittymänä pelille. Se päivittää jokaisen
26   * kortin klikkauksen jälkeen tilanteen pöydällä.
27   *
28   * @author liisapauliina
29   */
30  public class GraafinenKayttoliittyma extends JPanel implements ActionListener {
31  
32      private JFrame ikkuna;
33      private ArrayList<JButton> kysymysnapit;
34      private ArrayList<JButton> vastausnapit;
35      private Logiikka logiikka;
36      private Pelitilanne peli;
37      private KorttienArpoja arpoja;
38      private JPanel korttipaneeli;
39      private JPanel nappipaneeli;
40  
41      /**
42       * Konstuktorissa käyttöliittymälle annetaan logiikka. Lisäksi alustetaan
43       * kysymykset ja vastaukset, jotka koostuvat JButtoneista.
44       *
45       * @param logiikka on pelin logiikka.
46       */
47      public GraafinenKayttoliittyma(Logiikka logiikka) {
48          this.logiikka = logiikka;
49          this.peli = logiikka.getPelitilanne();
50          this.kysymysnapit = new ArrayList<JButton>();
51          this.vastausnapit = new ArrayList<JButton>();
52      }
53  
54      /**
55       * Metodi avaa peliikkunan ja asettaa JButtoneista koostuvat kysymykset ja
56       * vastaukset pöydälle.
57       */
58      public void run() {
59          ikkuna = new JFrame("Kertolaskumuistipeli");
60          ikkuna.setPreferredSize(new Dimension(800, 400));
61          ikkuna.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
62          this.lisaaKomponentit();
63          ikkuna.pack();
64          ikkuna.setVisible(true);
65      }
66  
67      /**
68       * Metodi, joka luo JButtonin jokaiselle kortille, asettaa sille kuuntelijan
69       * sekä lisää sen korttipaneeliin. Metodi luo myös uusi peli-nappulan ja
70       * lisää sen nappipaneeliin.
71       *
72       * @param container on laatikko, johon kortit asetetaan.
73       */
74      public void lisaaKomponentit() {
75  
76          GridLayout gl = new GridLayout(0, 10);
77          korttipaneeli = new JPanel(gl);
78          FlowLayout fl = new FlowLayout();
79          nappipaneeli = new JPanel(fl);
80          ikkuna.getContentPane().add(korttipaneeli, BorderLayout.CENTER);
81          ikkuna.getContentPane().add(nappipaneeli, BorderLayout.SOUTH);
82  
83          for (Kortti kortti : this.peli.getKysymykset()) {
84              JButton nappi = new JButton();
85              this.kysymysnapit.add(nappi);
86              KysymystenKuuntelija kuuntelija = new KysymystenKuuntelija(this.peli.getKysymykset().indexOf(kortti), this.logiikka, this);
87              nappi.addActionListener(kuuntelija);
88              nappi.setText(kortti.toString());
89              korttipaneeli.add(nappi);
90  
91          }
92          for (Kortti kortti : this.peli.getVastaukset()) {
93              JButton nappi = new JButton();
94              this.vastausnapit.add(nappi);
95              VastaustenKuuntelija kuuntelija = new VastaustenKuuntelija(this.peli.getVastaukset().indexOf(kortti), this.logiikka, this);
96              nappi.addActionListener(kuuntelija);
97              nappi.setText(kortti.toString());
98              korttipaneeli.add(nappi);
99          }
100 
101         JButton uusiPeli = new JButton("Valitse uusi peli");
102         uusiPeli.addActionListener(this);
103         nappipaneeli.add(uusiPeli);
104     }
105 
106     /**
107      * Metodi päivittää tekstit korteissa.
108      */
109     public void paivitaKortit() {
110 
111         for (int i = 0; i < this.kysymysnapit.size(); i++) {
112             kysymysnapit.get(i).setText(this.peli.getKysymykset().get(i).toString());
113         }
114         for (int i = 0; i < this.vastausnapit.size(); i++) {
115             vastausnapit.get(i).setText(this.peli.getVastaukset().get(i).toString());
116         }
117 
118     }
119 
120     public Logiikka getLogiikka() {
121         return logiikka;
122     }
123 
124     /**
125      * Metodi tekee ponnahdusikkunan, joka kysyy pelaajalta haluaako hän uuden
126      * pelin. Jos vastataan "K", käynnistetään uusi peli. Jos vastaus on "E", ei
127      * tehdä mitään. Jos vastaus on joku muu, kysytään uudelleen.
128      */
129     public void peliLoppui() {
130         String mitaTehdaan = PonnahdusIkkunanLuoja.kysy("Haluatko uuden pelin? K = Kyllä, E = En");
131         while (!mitaTehdaan.equals("K") && !mitaTehdaan.equals("E")) {
132             mitaTehdaan = PonnahdusIkkunanLuoja.kysy("Haluatko uuden pelin? K = Kyllä, E = En");
133         }
134         if (mitaTehdaan.equals("K")) {
135             PelinAloittaja aloittaja = new PelinAloittaja();
136             aloittaja.teeUusiPeli(aloittaja.kysyKertoja());
137         } else {
138             System.out.println("päätit lopettaa pelin");
139         }
140     }
141 
142     @Override
143     public void actionPerformed(ActionEvent e) {
144         peliLoppui();
145     }
146 }
