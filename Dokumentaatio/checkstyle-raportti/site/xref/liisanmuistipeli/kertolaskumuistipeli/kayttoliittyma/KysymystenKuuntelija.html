8   import java.awt.event.ActionEvent;
9   import java.awt.event.ActionListener;
10  import java.util.logging.Level;

12  import liisanmuistipeli.kertolaskumuistipeli.logiikka.Logiikka;

16  /**
17   * Luokka toimii kysymyskortin kuuntelijana ja välittää käskyn logiikalle, kun
18   * kysymyskortti valitaan. Käyttöliittymää pyydetään päivittämään kortit, kun
19   * kortti valitaan.
20   *
21   * @author liisapauliina
22   */
23  public class KysymystenKuuntelija implements ActionListener {
24  
25      private Logiikka logiikka;
26      private int paikka;
27      private GraafinenKayttoliittyma kali;
28  
29      /**
30       * Konstruktorissa kuuntelijalle annetaan kortin paikka pöydällä, pelin
31       * logiikka ja käyttöliittyma.
32       *
33       * @param paikka on kysymyskortin paikka pöydällä
34       * @param logiikka on pelin logiikka
35       * @param kali on pelin käyttöliittymä
36       */
37      public KysymystenKuuntelija(int paikka, Logiikka logiikka, GraafinenKayttoliittyma kali) {
38          this.logiikka = logiikka;
39          this.paikka = paikka;
40          this.kali = kali;
41      }
42  
43      @Override
44      public void actionPerformed(ActionEvent e) {
45          if (this.logiikka.saakoJoKaantaa()) {
46              this.logiikka.valitseKysymyskortti(this.paikka);
47              this.kali.paivitaKortit();
48          }
49      }
50  }
