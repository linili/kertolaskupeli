6   package liisanmuistipeli.kertolaskumuistipeli.logiikka;
7   
8   /**
9    * Luokka kuvaa yhtä muistipelin korttia. Kortilla on parametreina tiedot siitä,
10   * onko kortti löydetty tai käännetty näkyviin. Lisäksi kortilla on kertoja,
11   * joka on sama tietyn pelin korteilla. Kortilla on myös kerrottava, joka on
12   * muistipelin pareilla sama. Tyyppi on 0, jos kortti on vastauskortti ja 1, jos
13   * se on kysymyskortti.
14   *
15   * @author liisapauliina
16   */
17  public class Kortti {
18  
19      private boolean loydetty;
20      private boolean kaannetty;
21      private int kertoja;
22      private int kerrottava;
23      private int tyyppi;
24  
25      /**
26       * Kostruktorissa alustetaan kortin kertoja, kerrottava ja tyyppi. Alussa
27       * kortti on väärinpäin ja löytämättä.
28       *
29       * @param kertoja on pelaajalta saatu luku (1-9)
30       * @param kerrottava on parin identifioiva luku
31       * @param tyyppi kertoo onko vastaus (0) vai kysymys (1).
32       */
33      public Kortti(int kertoja, int kerrottava, int tyyppi) {
34          this.kertoja = kertoja;
35          this.kerrottava = kerrottava;
36          this.tyyppi = tyyppi;
37          this.loydetty = false;
38          this.kaannetty = false;
39      }
40  
41      public int getKertoja() {
42          return this.kertoja;
43      }
44  
45      public int getKerrottava() {
46          return this.kerrottava;
47      }
48  
49      public int getTyyppi() {
50          return this.tyyppi;
51      }
52  
53      /**
54       * Metodi asettaa kortin näkyviin.
55       */
56      public void nayta() {
57          kaannetty = true;
58      }
59  
60      /**
61       * Metodi asettaa kortin löydetyksi ja asettaa käännetyn arvoksi false.
62       */
63      public void poistaPoydalta() {
64          loydetty = true;
65          this.kaannetty = false;
66      }
67  
68      /**
69       * Metodi asettaa kortin pois näkyvistä
70       */
71      public void piiloon() {
72          kaannetty = false;
73      }
74  
75      /**
76       * Metodi palauttaa tiedon siitä, onko kortti käännettynä. Käännetty
77       * tarkoittaa tässä sitä, onko kortti valittuna yhden vuoron aikana.
78       *
79       * @return kaannetty eli tieto siitä, onko kortti valittuna.
80       */
81      public boolean onkoKaannetty() {
82          return kaannetty;
83      }
84  
85      /**
86       * Metodi palauttaa tiedon siitä, onko kortti jo löydetty
87       *
88       * @return löydetty, eli tieto siitä onko kortti löydetty
89       */
90      public boolean onkoLoydetty() {
91          return loydetty;
92      }
93  
94      @Override
95      public String toString() {
96  
97          if (this.kaannetty || this.loydetty) {
98  
99              if (this.tyyppi == 0) {
100 
101                 return Integer.toString(this.kerrottava * this.kertoja);
102             }
103             if (this.tyyppi == 1) {
104                 return Integer.toString(this.kertoja) + "x" + Integer.toString(this.kerrottava);
105             }
106         }
107         return "XXX";
108     }
109 
110 }
